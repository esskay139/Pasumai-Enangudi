function saveData() {

function showView(viewId){

/* =================== Products Rendering =================== */

function renderProducts(){
  const container = document.getElementById('products-grid');
  container.innerHTML = '';
  products.forEach(p => {
    const card = document.createElement('div');
    card.className='product-card';
    card.innerHTML=`
      <img src="${p.image}" alt="${p.name}"/>
      <div class="product-name">${p.name}</div>
      <div class="price">${settings.currencySymbol}${p.price}</div>
      <div class="quantity-controls">
        <button onclick="changeQuantity('${p.id}',-1)">-</button>
        <span id="qty-${p.id}">0</span>
        <button onclick="changeQuantity('${p.id}',1)">+</button>
      </div>
      <button class="add-to-cart" onclick="addToCart('${p.id}')">Add to Cart</button>
    `;
    container.appendChild(card);
  });
}

function changeQuantity(productId, delta){
  let item = document.getElementById(`qty-${productId}`);
  let qty = parseInt(item.textContent);
  qty += delta;
  if(qty<0) qty=0;
  item.textContent=qty;
}

function addToCart(productId){
  const product = products.find(p=>p.id==productId);
  const qtySpan = document.getElementById(`qty-${productId}`);
  const qty = parseInt(qtySpan.textContent);
  if(qty<=0) return;
  const cartItem = cart.find(c=>c.id==product.id);
  if(cartItem){
    cartItem.quantity += qty;
  }else{
    cart.push({id:product.id, name:product.name, price:product.price, quantity:qty});
  }
  // Reset qty selector
  qtySpan.textContent=0;
  alert(`${product.name} added to cart`);
  saveData();
}

/* =================== Cart Functions =================== */

function renderCart(){
  const tbody = document.querySelector('#cart-table tbody');
  tbody.innerHTML='';
  let totalQty=0, totalAmount=0;
  cart.forEach((item,index)=>{
    const row=document.createElement('tr');
    row.innerHTML=`
      <td>${item.name}</td>
      <td>
        <button onclick="updateCartQty(${index},-1)">-</button>
        ${item.quantity}
        <button onclick="updateCartQty(${index},1)">+</button>
      </td>
      <td>${settings.currencySymbol}${item.price}</td>
      <td>${settings.currencySymbol}${(item.price*item.quantity).toFixed(2)}</td>
      <td>
        <button onclick="removeFromCart(${index})">Remove</button>
      </td>
    `;
    tbody.appendChild(row);
    totalQty+=item.quantity;
    totalAmount+=(item.price*item.quantity);
  });
  document.getElementById('cart-total').innerHTML=`
    Total Items: ${totalQty} &nbsp; | &nbsp; Total: ${settings.currencySymbol}${totalAmount.toFixed(2)}
  `;
}

function updateCartQty(index, delta){
  if(index<0 || index>=cart.length) return;
  cart[index].quantity+=delta;
  if(cart[index].quantity<0) cart[index].quantity=0;
  if(cart[index].quantity===0) removeFromCart(index);
  renderCart();
  saveData();
}

function removeFromCart(index){
  cart.splice(index,1);
  renderCart();
  saveData();
}

function clearCart(){
  if(confirm('Clear entire cart?')){
    cart=[];
    renderCart();
    saveData();
  }
}

/* =================== Generate Bill =================== */

function generateBill(){
  if(cart.length===0){
    alert('Cart is empty');
    return;
  }
  const customerName = document.getElementById('cust-name').value.trim();
  const customerPhone = document.getElementById('cust-phone').value.trim();
  const paymentMethod = document.getElementById('payment-method').value;
  const notes = document.getElementById('cust-notes').value.trim();
  const billNumber = 'BILL'+Date.now();
  const dateTime = new Date();
  const dateStr = dateTime.toLocaleDateString();
  const timeStr = dateTime.toLocaleTimeString();

  const billDetails = {
    billNumber,
    date: dateStr,
    time: timeStr,
    customerName,
    customerPhone,
    paymentMethod,
    notes,
    items: [...cart],
    totalAmount: cart.reduce((a,c)=>a+c.price*c.quantity,0),
  };

  // Save bill
  bills.push(billDetails);
  // Save sales
  sales.push({date: dateStr, amount: billDetails.totalAmount, items: [...cart]});
  // Save to localStorage
  saveData();

  // Generate invoice display
  let invoiceHTML = `
    <h3>${settings.shopName}</h3>
    <p>${settings.slogan}</p>
    <p>Bill No: ${billNumber}<br/>Date: ${dateStr} ${timeStr}</p>
    <p>Customer: ${customerName}<br/>Phone: ${customerPhone}</p>
    <p>Payment: ${paymentMethod}</p>
    <table style="width:100%; border-collapse: collapse;">
      <thead>
        <tr style="border:1px solid #ccc;"><th style="border:1px solid #ccc;">Product</th><th style="border:1px solid #ccc;">Qty</th><th style="border:1px solid #ccc;">Price</th><th style="border:1px solid #ccc;">Total</th></tr>
      </thead>
      <tbody>
  `;
  cart.forEach(item => {
    invoiceHTML+=`
      <tr style="border:1px solid #ccc;">
        <td style="border:1px solid #ccc;">${item.name}</td>
        <td style="border:1px solid #ccc;">${item.quantity}</td>
        <td style="border:1px solid #ccc;">${settings.currencySymbol}${item.price}</td>
        <td style="border:1px solid #ccc;">${settings.currencySymbol}${(item.price*item.quantity).toFixed(2)}</td>
      </tr>
    `;
  });
  invoiceHTML+=`
      </tbody>
    </table>
    <h4>Total: ${settings.currencySymbol}${billDetails.totalAmount.toFixed(2)}</h4>
    <p>Notes: ${notes}</p>
    ${settings.invoiceFooter}
  `;

  document.getElementById('invoice-details').innerHTML=invoiceHTML;

  // Show billing section
  showView('billing-section');

  // Clear cart
  cart=[];
  renderCart();
}

function generateYearlyReport(){
  const year = new Date().getFullYear();
  const yearlySales = sales.filter(s => s.date.endsWith(year));
  displaySalesReport(yearlySales, 'Yearly Sales Report');
}

function displaySalesReport(salesData, title){
  const div = document.getElementById('report-output');
  let html='';

  html+=`<h3>${title}</h3>`;
  if(salesData.length===0){
    html+='<p>No sales data.</p>';
  }else{
    let totalSales=0;
    let totalItems=0;
    salesData.forEach(s => {
      totalSales+=s.amount;
      totalItems+=s.items.reduce((a,c)=>a+c.quantity,0);
    });
    html+=`<p>Total Bills: ${salesData.length}</p>`;
    html+=`<p>Total Items Sold: ${totalItems}</p>`;
    html+=`<p>Total Sales: ${settings.currencySymbol}${totalSales.toFixed(2)}</p>`;
    // Best Product
    const productCounts = {};
    salesData.forEach(s => {
      s.items.forEach(i => {
        productCounts[i.name] = (productCounts[i.name] || 0)+i.quantity;
      });
    });
    const bestProduct = Object.entries(productCounts).sort((a,b)=>b[1]-a[1])[0];
    html+=`<p>Best Selling Product: ${bestProduct ? bestProduct[0] : 'N/A'} (${bestProduct ? bestProduct[1] : 0} units)</p>`;
  }
  div.innerHTML=html;
}

/* =================== Dashboard =================== */

let chartInstance = null;

function updateDashboard(){
  const ctx = document.getElementById('salesChart').getContext('2d');

  const today = new Date();
  const labels = [];
  const data = [];

  // Generate last 7 days data
  for(let i=6;i>=0;i--){
    const date = new Date(today);
    date.setDate(today.getDate()-i);
    const dateStr = date.toLocaleDateString();
    labels.push(dateStr);
    const daySales = sales.filter(s => s.date===dateStr);
    const total = daySales.reduce((a,c)=>a+c.amount,0);
    data.push(total);
  }

  if(chartInstance) chartInstance.destroy();

  chartInstance = new Chart(ctx, {
    type:'line',
    data:{
      labels,
      datasets:[{
        label:'Sales (last 7 days)',
        data,
        borderColor:'var(--dark-green)',
        backgroundColor:'rgba(0,100,0,0.2)',
        fill:true,
        tension:0.3,
      }]
    },
    options:{
      responsive:true,
      plugins: {
        legend: { display:false }
      }
    }
  });
}

}

/* =================== Backup & Restore =================== */

function backupData(){
  const dataStr=JSON.stringify({products,cart,sales,bills,settings,adminPassword});
  const blob=new Blob([dataStr],{type:'application/json'});
  const url=URL.createObjectURL(blob);
  const a=document.createElement('a');
  a.href=url;
  a.download='pasumai_backup.json';
  a.click();
}

function restoreData(){
  const input=document.createElement('input');
  input.type='file';
  input.accept='application/json';
  input.onchange=function(e){
    const file=e.target.files[0];
    const reader=new FileReader();
    reader.onload=function(){
      try{
        const data=JSON.parse(reader.result);
        products=data.products || products;
        cart=data.cart || [];
        sales=data.sales || [];
        bills=data.bills || [];
        settings=data.settings || settings;
        adminPassword=data.adminPassword || 'admin123';
        saveData();
        alert('Data restored.');
        showView('home-screen');
      }catch(e){
        alert('Invalid file');
      }
    }
    reader.readAsText(file);
  }
  input.click();
}
